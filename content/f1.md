<script>
fetch('/f1/index.json')
  .then(r => r.json())
  .then(index => {
    const selector = document.getElementById('race-selector-container');
    selector.innerHTML = `
      <div style="margin-bottom:20px;">
        <label for="race-select"><strong>Select Session: </strong></label>
        <select id="race-select" onchange="loadRace(this.value)" style="padding:4px 8px;">
          ${index.slice().reverse().map(r => {
            const isSprint = r.session_type === 'Sprint';
            const label = isSprint
              ? `${r.race} Sprint ${r.year}`
              : `${r.race} GP ${r.year}`;
            return `<option value="${r.file_path}">${label}</option>`;
          }).join('')}
        </select>
      </div>
    `;

    const latest = index[index.length - 1];
    loadRace(latest.file_path);
  });

function loadRace(filePath) {
  fetch(`/f1/${filePath}`)
    .then(r => r.json())
    .then(data => renderRace(data));
}

function renderRace(data) {
  const el = document.getElementById('f1-results');
  const isSprint = data.session_type === 'Sprint';
  const sessionLabel = isSprint ? 'Sprint' : 'Grand Prix';
  const sessionBadge = isSprint
    ? '<span style="display:inline-block;margin-left:10px;padding:2px 8px;background:#e8002d;color:#fff;font-size:0.75em;border-radius:4px;vertical-align:middle;">SPRINT</span>'
    : '';

  el.innerHTML = `
    <h2>${data.race} ${sessionLabel} ${data.year} ${sessionBadge}</h2>
    <p>${data.date} &mdash; ${data.circuit}</p>
    <p>🌡️ Air: ${data.weather.start.air_temperature}°C &nbsp; Track: ${data.weather.start.track_temperature}°C &nbsp; Humidity: ${data.weather.start.humidity}%</p>

    <h3>Results</h3>
    <table>
      <thead>
        <tr><th>Pos</th><th>Driver</th><th>Team</th><th>Laps</th></tr>
      </thead>
      <tbody>
        ${data.results.map(r => `
          <tr>
            <td>${r.position}</td>
            <td>${r.full_name} ${data.fastest_lap && data.fastest_lap.driver_number === r.driver_number ? '⚡' : ''}</td>
            <td>${r.team}</td>
            <td>${r.laps_completed}</td>
          </tr>
        `).join('')}
      </tbody>
    </table>

    <h3>Fastest Lap</h3>
    <p>⚡ ${data.fastest_lap.full_name} &mdash; ${data.fastest_lap.lap_duration_formatted} on lap ${data.fastest_lap.lap_number}</p>

    ${isSprint ? '' : `
    <h3>Tyre Strategy</h3>
    <h4>Tap, click, or hover over bars for Pit Details</h4>
    `}
    ${!isSprint ? buildStrategy(data) : buildSprintStrategy(data)}
  `;
}

function showPitInfo(el, text) {
  const existing = document.getElementById('pit-popup');
  if (existing) {
    existing.remove();
    if (existing.dataset.source === el.dataset.id) return;
  }

  const id = Math.random().toString(36).slice(2);
  el.dataset.id = id;

  const popup = document.createElement('div');
  popup.id = 'pit-popup';
  popup.dataset.source = id;
  popup.style.cssText = `
    background:#222;color:#fff;padding:6px 10px;border-radius:4px;
    font-size:0.8em;margin-top:8px;display:inline-block;
    max-width:100%;
  `;
  popup.textContent = text;

  document.getElementById('f1-results').appendChild(popup);
}

function buildSprintStrategy(data) {
  const compounds = {
    SOFT: '#e8002d',
    MEDIUM: '#ffd700',
    HARD: '#ffffff',
    INTERMEDIATE: '#43b02a',
    WET: '#0067ff',
    UNKNOWN: '#888'
  };

  if (!data.strategy || data.strategy.length === 0) {
    return '<p style="color:#888;font-size:0.9em;">No tyre strategy data available for this sprint.</p>';
  }

  const drivers = [...new Set(data.strategy.map(s => s.driver_number))];
  const totalLaps = Math.max(...data.results.map(r => r.laps_completed));

  const rows = drivers.map(num => {
    const stints = data.strategy.filter(s => s.driver_number === num);
    const driver = data.results.find(r => r.driver_number === num);
    const bars = stints.map(s => {
      const width = ((s.lap_end - s.lap_start + 1) / totalLaps * 100).toFixed(1);
      const color = compounds[s.compound] || '#888';
      const border = s.compound === 'HARD'
        ? 'border: 1px solid #555; border-right: 2px solid #333;'
        : 'border-right: 2px solid #333;';
      const title = `${s.compound} L${s.lap_start}-${s.lap_end}`;
      return `<div title="${title}" style="display:inline-block;width:${width}%;background:${color};${border}height:20px;"></div>`;
    }).join('');

    return `
      <div style="display:flex;align-items:center;margin:4px 0;gap:8px;">
        <span style="width:40px;font-size:0.85em;">${driver ? 'P' + driver.position : ''}</span>
        <span style="width:35px;font-size:0.85em;">${stints[0]?.acronym || ''}</span>
        <div style="flex:1;">${bars}</div>
      </div>`;
  }).join('');

  const legend = Object.entries(compounds).map(([name, color]) => `
    <span style="display:inline-flex;align-items:center;gap:4px;margin-right:12px;font-size:0.8em;">
      <div style="width:14px;height:14px;background:${color};border:1px solid #555;"></div>${name}
    </span>`).join('');

  return `
    <h3>Sprint Tyre Strategy</h3>
    <div style="margin-bottom:8px;">${legend}</div>
    <div>${rows}</div>`;
}

function buildStrategy(data) {
  const compounds = {
    SOFT: '#e8002d',
    MEDIUM: '#ffd700',
    HARD: '#ffffff',
    INTERMEDIATE: '#43b02a',
    WET: '#0067ff',
    UNKNOWN: '#888'
  };

  const drivers = [...new Set(data.strategy.map(s => s.driver_number))];
  const totalLaps = Math.max(...data.results.map(r => r.laps_completed));

  const rows = drivers.map(num => {
    const stints = data.strategy.filter(s => s.driver_number === num);
    const driver = data.results.find(r => r.driver_number === num);
    const bars = stints.map((s, idx) => {
      const width = ((s.lap_end - s.lap_start + 1) / totalLaps * 100).toFixed(1);
      const color = compounds[s.compound] || '#888';
      const border = s.compound === 'HARD'
        ? 'border: 1px solid #555; border-right: 2px solid #333;'
        : 'border-right: 2px solid #333;';

      const pit = data.pit_stops?.find(p =>
        p.driver_number === num &&
        (p.lap_number === s.lap_end ||
         p.lap_number === s.lap_end + 1 ||
         p.lap_number === s.lap_end - 1)
      );
      const pitTitle = pit
        ? `${s.compound} L${s.lap_start}-${s.lap_end} | Pit lap ${pit.lap_number}: lane ${pit.lane_duration}s${pit.stop_duration ? ' / stop ' + pit.stop_duration + 's' : ''}${pit.abnormal ? ' ⚠️ abnormal' : ''}`
        : `${s.compound} L${s.lap_start}-${s.lap_end}`;

      return `<div
        onclick="showPitInfo(this, '${pitTitle.replace(/'/g, '&apos;')}')"
        title="${pitTitle}"
        style="display:inline-block;width:${width}%;background:${color};${border}height:20px;cursor:pointer;">
      </div>`;
    }).join('');

    return `
      <div style="display:flex;align-items:center;margin:4px 0;gap:8px;">
        <span style="width:40px;font-size:0.85em;">${driver ? 'P' + driver.position : ''}</span>
        <span style="width:35px;font-size:0.85em;">${stints[0]?.acronym || ''}</span>
        <div style="flex:1;">${bars}</div>
      </div>`;
  }).join('');

  const legend = Object.entries(compounds).map(([name, color]) => `
    <span style="display:inline-flex;align-items:center;gap:4px;margin-right:12px;font-size:0.8em;">
      <div style="width:14px;height:14px;background:${color};border:1px solid #555;"></div>${name}
    </span>`).join('');

  return `<div style="margin-bottom:8px;">${legend}</div><div>${rows}</div>`;
}
