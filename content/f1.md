<script>
fetch('/f1/index.json')
  .then(r => r.json())
  .then(index => {
    // Build race selector
    const selector = document.getElementById('race-selector-container');
    selector.innerHTML = `
      <div style="margin-bottom:20px;">
        <label for="race-select"><strong>Select Race: </strong></label>
        <select id="race-select" onchange="loadRace(this.value)" style="padding:4px 8px;">
          ${index.slice().reverse().map(r => `
            <option value="${r.slug}">${r.race} GP ${r.year}</option>
          `).join('')}
        </select>
      </div>
    `;

    // Load most recent race by default
    const latest = index[index.length - 1];
    loadRace(latest.slug);
  });

function loadRace(slug) {
  fetch(`/f1/${slug}/summary.json`)
    .then(r => r.json())
    .then(data => renderRace(data));
}

function renderRace(data) {
  console.log('pit_stops sample:', JSON.stringify(data.pit_stops?.slice(0,3)));
  console.log('strategy sample:', JSON.stringify(data.strategy?.slice(0,3)));
  const el = document.getElementById('f1-results');
  el.innerHTML = `
    <h2>${data.race} Grand Prix ${data.year}</h2>
    <p>${data.date} &mdash; ${data.circuit}</p>
    <p>🌡️ Air: ${data.weather.start.air_temperature}°C &nbsp; Track: ${data.weather.start.track_temperature}°C &nbsp; Humidity: ${data.weather.start.humidity}%</p>

    <h3>Race Results</h3>
    <table>
      <thead>
        <tr><th>Pos</th><th>Driver</th><th>Team</th><th>Laps</th></tr>
      </thead>
      <tbody>
        ${data.results.map(r => `
          <tr>
            <td>${r.position}</td>
            <td>${r.full_name} ${data.fastest_lap && data.fastest_lap.driver_number === r.driver_number ? '⚡' : ''}</td>
            <td>${r.team}</td>
            <td>${r.laps_completed}</td>
          </tr>
        `).join('')}
      </tbody>
    </table>

    <h3>Fastest Lap</h3>
    <p>⚡ ${data.fastest_lap.full_name} &mdash; ${data.fastest_lap.lap_duration_formatted} on lap ${data.fastest_lap.lap_number}</p>

    <h3>Tyre Strategy</h3>
    ${buildStrategy(data)}
  `;
}

function buildStrategy(data) {
  const compounds = {
    SOFT: '#e8002d',
    MEDIUM: '#ffd700',
    HARD: '#ffffff',
    INTERMEDIATE: '#43b02a',
    WET: '#0067ff',
    UNKNOWN: '#888'
  };

  const drivers = [...new Set(data.strategy.map(s => s.driver_number))];
  const totalLaps = Math.max(...data.results.map(r => r.laps_completed));

  const rows = drivers.map(num => {
    const stints = data.strategy.filter(s => s.driver_number === num);
    const driver = data.results.find(r => r.driver_number === num);
    const bars = stints.map((s, idx) => {
      const width = ((s.lap_end - s.lap_start + 1) / totalLaps * 100).toFixed(1);
      const color = compounds[s.compound] || '#888';
      const border = s.compound === 'HARD'
        ? 'border: 1px solid #555; border-right: 2px solid #333;'
        : 'border-right: 2px solid #333;';

      // Find pit stop at end of this stint
      const pit = data.pit_stops?.find(p =>
        p.driver_number === num && p.lap_number === s.lap_end + 1
      );
      const pitTitle = pit
        ? `${s.compound} L${s.lap_start}-${s.lap_end} | Pit lap ${pit.lap_number}: lane ${pit.lane_duration}s${pit.stop_duration ? ' / stop ' + pit.stop_duration + 's' : ''}${pit.abnormal ? ' ⚠️ abnormal' : ''}`
        : `${s.compound} L${s.lap_start}-${s.lap_end}`;
        
      return `<div title="${pitTitle}" style="display:inline-block;width:${width}%;background:${color};${border}height:20px;cursor:pointer;"></div>`;
    }).join('');

    return `
      <div style="display:flex;align-items:center;margin:4px 0;gap:8px;">
        <span style="width:40px;font-size:0.85em;">${driver ? 'P' + driver.position : ''}</span>
        <span style="width:35px;font-size:0.85em;">${stints[0]?.acronym || ''}</span>
        <div style="flex:1;">${bars}</div>
      </div>`;
  }).join('');

  // Legend
  const legend = Object.entries(compounds).map(([name, color]) => `
    <span style="display:inline-flex;align-items:center;gap:4px;margin-right:12px;font-size:0.8em;">
      <div style="width:14px;height:14px;background:${color};border:1px solid #555;"></div>${name}
    </span>`).join('');

  return `<div style="margin-bottom:8px;">${legend}</div><div>${rows}</div>`;
}
